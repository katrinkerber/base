Standard form

03modules/forms.scss

JS-driven validation is powered by the jQuery Validate Plugin.

All validation rules are applied to the form field via attributes.

The required attribute on a field triggers either the JS validation, or the default browser validation if JS is disabled.

The data-msg-required attribute defines the error message that appears when the field is invalid.

Special rules can also be added for special fields, i.e. email, url, numbers, minlength etc.. Check the documentation here or this post for built-in and additional validation rules.


<form class="js-validate" method="post" autocomplete="off">
  <fieldset class="form__fields">
    <div class="form__fields__group">
      <div class="form__field">
        <label for="name">
          Your name
        </label>
        <input
          name="name"
          id="name"
          type="text"

          aria-required="true"
          required
          data-msg-required="Please enter your name."
        >
      </div>

      <div class="form__field form__field--inline">
        <label for="number">
          Telephone number
        </label>
        <input
          name="number"
          id="number"
          type="tel"

          aria-required="true"
          required
          data-msg-required="Please add a number."
          data-rule-digits="true"
          data-msg-digits="Please enter only numerical characters."
        >
      </div>
    </div>

    <div class="form__fields__group">
      <div class="form__field">
        <label for="email">
          Email address
        </label>
        <input
          name="email"
          id="email"
          type="email"

          aria-required="true"
          required
          data-msg-required="Please enter your email."
          data-rule-email="true"
          data-msg-email="Please enter a valid email address."
        >
      </div>

      <div class="form__field">
        <label for="company">
          Company name
        </label>
        <input
          name="company"
          id="company"
          type="text"

          aria-required="true"
          required
          data-msg-required="Please enter your name."
        >
      </div>
    </div>

    <div class="form__field">
      <label for="message">
        Your message
      </label>
      <textarea
        name="message"
        id="message"
        rows="1"
      ></textarea>
    </div>
  </fieldset>

  <button class="form__btn btn btn--white" type="submit">Send</button>
</form>
